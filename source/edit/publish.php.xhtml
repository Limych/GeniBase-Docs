function trim_text($text, $max_len = 70)
{
    $text = trim($text);
    if (mb_strlen($text) > $max_len)
        $text = preg_replace('/\s+\w*$/uS', '', mb_substr($text, 0, $max_len)) . '…';
    return $text;
}
